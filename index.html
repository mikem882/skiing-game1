let current = 0;
let activeCurveball = null;

const curveballs = [
  { text:"Your ski pole snaps in half!",
    choices:[
      { text:"Keep skiing", enjoyment:-5, next:5 },
      { text:"Get replacement", enjoyment:-2, next:5 }
    ]},
  { text:"Chair lifts stop working!",
    choices:[
      { text:"Wait it out", enjoyment:-4, next:5 },
      { text:"Hike uphill", enjoyment:-8, next:5 }
    ]},
  { text:"You fall into a freezing river!",
    choices:[
      { text:"Warm up inside", enjoyment:-6, next:5 },
      { text:"Keep skiing soaked", enjoyment:-10, next:5 }
    ]},
  { text:"You drop your phone off the lift!",
    choices:[
      { text:"Search for it", enjoyment:-5, next:5 },
      { text:"Accept the loss", enjoyment:-3, next:5 }
    ]}
];

function getEnding(score){
  if(score<=9) return "Your day was mid 😐";
  if(score<=25) return "Rough day 😬";
  if(score<=50) return "Decent ski day 🙂";
  if(score<=75) return "Awesome day 🤩";
  return "Best ski day ever!! 🏆";
}

function showScenario(){
  const story=document.getElementById("story");
  const choices=document.getElementById("choices");
  const meter=document.getElementById("meter");

  meter.textContent=`Enjoyment: ${enjoyment}`;
  choices.innerHTML="";
  const scene=scenarios[current];
  story.textContent=scene.text;

  if(scene.end){
    const result=document.createElement("h2");
    result.textContent=getEnding(enjoyment);
    story.appendChild(result);

    const restart=document.createElement("button");
    restart.textContent="Play Again";
    restart.onclick=()=>{ enjoyment=50; current=0; activeCurveball=null; showScenario(); };
    choices.appendChild(restart);
    return;
  }

  scene.choices.forEach(choice=>{
    const btn=document.createElement("button");
    btn.textContent=choice.text;
    btn.onclick=()=>{
      enjoyment+=choice.enjoyment;
      enjoyment=Math.max(0,Math.min(100,enjoyment));

      if(current===4 && activeCurveball===null){
        activeCurveball=curveballs[Math.floor(Math.random()*curveballs.length)];
        showCurveball();
        return;
      }

      current=choice.next;
      showScenario();
    };
    choices.appendChild(btn);
  });
}

function showCurveball(){
  const story=document.getElementById("story");
  const choices=document.getElementById("choices");
  story.textContent=activeCurveball.text;
  choices.innerHTML="";

  activeCurveball.choices.forEach(choice=>{
    const btn=document.createElement("button");
    btn.textContent=choice.text;
    btn.onclick=()=>{
      enjoyment+=choice.enjoyment;
      current=choice.next;
      showScenario();
    };
    choices.appendChild(btn);
  });
}

showScenario();
